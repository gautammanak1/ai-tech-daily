## Company Overview

Codeium is a free AI-powered toolkit designed to enhance the coding experience for developers across the entire software development lifecycle. Developed by Exafunction and founded in 2021 by Varun Mohan and Douglas Chen—both MIT graduates and childhood friends—Codeium has rapidly evolved from a simple code completion tool into a comprehensive AI coding platform.

### Mission and Vision

At its core, Codeium's mission is to democratize AI-powered development tools by making them accessible to all developers without paywalls. The company believes that AI assistance shouldn't be a premium feature reserved for enterprise teams but rather a fundamental capability available to every coder, from students to senior engineers at Fortune 500 companies.

### Founding Story

The founding duo brings impressive technical credentials to the table. Douglas Chen previously worked at Meta, where he helped build software tools for VR headsets including the Oculus Quest. Varun Mohan, serving as CEO, has led the company through rapid growth while maintaining a lean operational structure. Their MIT background and shared history have enabled them to build a company culture focused on technical excellence and rapid iteration.

### Using Codeium Chat for Complex Tasks

Codeium Chat shines when you need to tackle complex development tasks. Here's an example of using it to implement a complete feature:

```typescript
// Prompt to Codeium Chat:
// "Create a React hook for managing WebSocket connections with auto-reconnect,
//  message queuing, and event handling. Include TypeScript types and error handling."

// Codeium Chat response:

import { useState, useEffect, useCallback, useRef } from 'react';

interface WebSocketHookOptions {
  url: string;
  reconnectInterval?: number;
  maxReconnectAttempts?: number;
  onMessage?: (data: any) => void;
  onError?: (error: Event) => void;
  onOpen?: (event: Event) => void;
  onClose?: (event: CloseEvent) => void;
}

interface WebSocketState {
  isConnected: boolean;
  isReconnecting: boolean;
  reconnectAttempts: number;
  lastError: Event | null;
}

interface QueuedMessage {
  data: string;
  timestamp: number;
}

export function useWebSocket(options: WebSocketHookOptions) {
  const {
    url,
    reconnectInterval = 3000,
    maxReconnectAttempts = 5,
    onMessage,
    onError,
    onOpen,
    onClose,
  } = options;

  const [state, setState] = useState<WebSocketState>({
    isConnected: false,
    isReconnecting: false,
    reconnectAttempts: 0,
    lastError: null,
  });

  const wsRef = useRef<WebSocket | null>(null);
  const messageQueue = useRef<QueuedMessage[]>([]);
  const reconnectTimeoutRef = useRef<NodeJS.Timeout | null>(null);

  const connect = useCallback(() => {
    try {
      wsRef.current = new WebSocket(url);

      wsRef.current.onopen = (event) => {
        setState(prev => ({
          ...prev,
          isConnected: true,
          isReconnecting: false,
          reconnectAttempts: 0,
          lastError: null,
        }));
        onOpen?.(event);

        // Send queued messages
        while (messageQueue.current.length > 0) {
          const message = messageQueue.current.shift();
          if (message && wsRef.current?.readyState === WebSocket.OPEN) {
            wsRef.current.send(message.data);
          }
        }
      };

      wsRef.current.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data);
          onMessage?.(data);
        } catch {
          onMessage?.(event.data);
        }
      };

      wsRef.current.onerror = (event) => {
        setState(prev => ({ ...prev, lastError: event }));
        onError?.(event);
      };

      wsRef.current.onclose = (event) => {
        setState(prev => ({
          ...prev,
          isConnected: false,
        }));
        onClose?.(event);

        // Attempt reconnection if not closed intentionally
        if (!event.wasClean && state.reconnectAttempts < maxReconnectAttempts) {
          setState(prev => ({ ...prev, isReconnecting: true }));
          reconnectTimeoutRef.current = setTimeout(() => {
            setState(prev => ({
              ...prev,
              reconnectAttempts: prev.reconnectAttempts + 1,
            }));
            connect();
          }, reconnectInterval);
        }
      };
    } catch (error) {
      onError?.(error as Event);
    }
  }, [url, reconnectInterval, maxReconnectAttempts, onMessage, onError, onOpen, onClose]);

  const disconnect = useCallback(() => {
    if (reconnectTimeoutRef.current) {
      clearTimeout(reconnectTimeoutRef.current);
    }
    if (wsRef.current) {
      wsRef.current.close();
      wsRef.current = null;
    }
    setState({
      isConnected: false,
      isReconnecting: false,
      reconnectAttempts: 0,
      lastError: null,
    });
  }, []);

  const send = useCallback((data: any) => {
    const message = typeof data === 'string' ? data : JSON.stringify(data);
    
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(message);
    } else {
      // Queue message for when connection is established
      messageQueue.current.push({
        data: message,
        timestamp: Date.now(),
      });
    }
  }, []);

  useEffect(() => {
    connect();
    return () => disconnect();
  }, [connect, disconnect]);

  return {
    ...state,
    send,
    connect,
    disconnect,
  };
}

// Usage example in a component
function ChatComponent() {
  const { isConnected, send, lastError } = useWebSocket({
    url: 'wss://api.example.com/chat',
    reconnectInterval: 5000,
    maxReconnectAttempts: 10,
    onMessage: (data) => {
      console.log('Received:', data);
      // Handle incoming messages
    },
    onError: (error) => {
      console.error('WebSocket error:', error);
    },
  });

  const handleSendMessage = (message: string) => {
    send({ type: 'chat', text: message });
  };

  return (
    <div>
      <div>Connection Status: {isConnected ? 'Connected' : 'Disconnected'}</div>
      {lastError && <div>Error: {lastError.toString()}</div>}
      {/* Chat UI implementation */}
    </div>
  );
}
```

# "Find all database query functions that don't use parameterized queries

    return cursor.fetchone()

    SELECT * FROM products 
    WHERE name LIKE '%{name}%'

    """

from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker
from typing import List, Optional

class DatabaseService:
    def __init__(self, connection_string: str):
        self.engine = create_engine(connection_string)
        self.SessionLocal = sessionmaker(bind=self.engine)
    
    def get_user_by_email(self, email: str) -> Optional[dict]:
        """Retrieve a user by email using parameterized query."""
        with self.SessionLocal() as session:
            result = session.execute(
                text("SELECT * FROM users WHERE email = :email"),
                {"email": email}
            )
            return result.fetchone()._asdict() if result.rowcount > 0 else None
    
    def search_products(
        self, 
        name: str, 
        category: Optional[str] = None
    ) -> List[dict]:
        """Search products with optional category filter."""
        with self.SessionLocal() as session:
            query = text("""
                SELECT * FROM products 
                WHERE name LIKE :name
                AND (:category IS NULL OR category = :category)
            """)
            result = session.execute(
                query,
                {
                    "name": f"%{name}%",
                    "category": category
                }
            )
            return [row._asdict() for row in result.fetchall()]
    
    def create_user(self, user_data: dict) -> dict:
        """Create a new user with all fields properly escaped."""
        with self.SessionLocal() as session:
            result = session.execute(
                text("""
                    INSERT INTO users (name, email, role, created_at)
                    VALUES (:name, :email, :role, NOW())
                    RETURNING *
                """),
                user_data
            )
            session.commit()
            return result.fetchone()._asdict()
